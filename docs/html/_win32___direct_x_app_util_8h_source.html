127  static LRESULT CALLBACK WindowProc(_In_ HWND hWnd, _In_ UINT Msg, _In_ WPARAM wParam, _In_ LPARAM lParam)
128  {
129  auto p = reinterpret_cast<DirectX11 *>(GetWindowLongPtr(hWnd, 0));
130  switch (Msg)
131  {
132  case WM_KEYDOWN:
133  p->Key[wParam] = true;
134  break;
135  case WM_KEYUP:
136  p->Key[wParam] = false;
137  break;
138  case WM_DESTROY:
139  p->Running = false;
140  break;
141  default:
142  return DefWindowProcW(hWnd, Msg, wParam, lParam);
143  }
144  if ((p->Key['Q'] && p->Key[VK_CONTROL]) || p->Key[VK_ESCAPE])
145  {
146  p->Running = false;
147  }
148  return 0;
149  }

208  bool InitDevice(int vpW, int vpH, const LUID* pLuid, bool windowed = true)
209  {
210  WinSizeW = vpW;
211  WinSizeH = vpH;
212 
213  RECT size = { 0, 0, vpW, vpH };
214  AdjustWindowRect(&size, WS_OVERLAPPEDWINDOW, false);
215  const UINT flags = SWP_NOMOVE | SWP_NOZORDER | SWP_SHOWWINDOW;
216  if (!SetWindowPos(Window, nullptr, 0, 0, size.right - size.left, size.bottom - size.top, flags))
217  return false;
218 
219  IDXGIFactory * DXGIFactory = nullptr;
220  HRESULT hr = CreateDXGIFactory1(__uuidof(IDXGIFactory), (void**)(&DXGIFactory));
221  VALIDATE((hr == ERROR_SUCCESS), "CreateDXGIFactory1 failed");
222 
223  IDXGIAdapter * Adapter = nullptr;
224  for (UINT iAdapter = 0; DXGIFactory->EnumAdapters(iAdapter, &Adapter) != DXGI_ERROR_NOT_FOUND; ++iAdapter)
225  {
226  DXGI_ADAPTER_DESC adapterDesc;
227  Adapter->GetDesc(&adapterDesc);
228  if ((pLuid == nullptr) || memcmp(&adapterDesc.AdapterLuid, pLuid, sizeof(LUID)) == 0)
229  break;
230  Release(Adapter);
231  }
232 
233  auto DriverType = Adapter ? D3D_DRIVER_TYPE_UNKNOWN : D3D_DRIVER_TYPE_HARDWARE;
234  hr = D3D11CreateDevice(Adapter, DriverType, 0, 0, 0, 0, D3D11_SDK_VERSION, &Device, 0, &Context);
235  Release(Adapter);
236  VALIDATE((hr == ERROR_SUCCESS), "D3D11CreateDevice failed");
237 
238  // Create swap chain
239  DXGI_SWAP_CHAIN_DESC scDesc;
240  memset(&scDesc, 0, sizeof(scDesc));
241  scDesc.BufferCount = 2;
242  scDesc.BufferDesc.Width = WinSizeW;
243  scDesc.BufferDesc.Height = WinSizeH;
244  scDesc.BufferDesc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
245  scDesc.BufferDesc.RefreshRate.Denominator = 1;
246  scDesc.BufferUsage = DXGI_USAGE_RENDER_TARGET_OUTPUT;
247  scDesc.OutputWindow = Window;
248  scDesc.SampleDesc.Count = 1;
249  scDesc.Windowed = windowed;
250  scDesc.SwapEffect = DXGI_SWAP_EFFECT_SEQUENTIAL;
251  hr = DXGIFactory->CreateSwapChain(Device, &scDesc, &SwapChain);
252  Release(DXGIFactory);
253  VALIDATE((hr == ERROR_SUCCESS), "CreateSwapChain failed");
254 
255  // Create backbuffer
256  SwapChain->GetBuffer(0, __uuidof(ID3D11Texture2D), (void**)&BackBuffer);
257  hr = Device->CreateRenderTargetView(BackBuffer, NULL, &BackBufferRT);
258  VALIDATE((hr == ERROR_SUCCESS), "CreateRenderTargetView failed");
259 
260  // Main depth buffer
261  MainDepthBuffer = new DepthBuffer(Device, WinSizeW, WinSizeH);
262  Context->OMSetRenderTargets(1, &BackBufferRT, MainDepthBuffer->TexDsv);
263 
264  // Buffer for shader constants
265  UniformBufferGen = new DataBuffer(Device, D3D11_BIND_CONSTANT_BUFFER, NULL, UNIFORM_DATA_SIZE);
266  Context->VSSetConstantBuffers(0, 1, &UniformBufferGen->D3DBuffer);
267 
268  // Set max frame latency to 1
269  IDXGIDevice1* DXGIDevice1 = nullptr;
270  hr = Device->QueryInterface(__uuidof(IDXGIDevice1), (void**)&DXGIDevice1);
271  DXGIDevice1->SetMaximumFrameLatency(1);
272  Release(DXGIDevice1);
273  VALIDATE((hr == ERROR_SUCCESS), "QueryInterface failed");
274 
275  return true;
276  }

297  {
298  MSG msg;
299  while (PeekMessage(&msg, NULL, 0U, 0U, PM_REMOVE))
300  {
301  TranslateMessage(&msg);
302  DispatchMessage(&msg);
303  }
304  // This is to provide a means to terminate after a maximum number of frames
305  // to facilitate automated testing
306  #ifdef MAX_FRAMES_ACTIVE
307  if (maxFrames > 0)
308  {
309  if (--maxFrames <= 0)
310  Running = false;
311  }
312  #endif
313  return Running;

402  void FillTexture(DWORD * pix)
403  {
404  //Make local ones, because will be reducing them
405  int sizeW = SizeW;
406  int sizeH = SizeH;
407  for (int level = 0; level < MipLevels; level++)
408  {
409  DIRECTX.Context->UpdateSubresource(Tex, level, NULL, (unsigned char *)pix, sizeW * 4, sizeH * 4);
410 
411  for (int j = 0; j < (sizeH & ~1); j += 2)
412  {
413  uint8_t* psrc = (uint8_t *)pix + (sizeW * j * 4);
414  uint8_t* pdest = (uint8_t *)pix + (sizeW * j);
415  for (int i = 0; i < sizeW >> 1; i++, psrc += 8, pdest += 4)
416  {
417  pdest[0] = (((int)psrc[0]) + psrc[4] + psrc[sizeW * 4 + 0] + psrc[sizeW * 4 + 4]) >> 2;
418  pdest[1] = (((int)psrc[1]) + psrc[5] + psrc[sizeW * 4 + 1] + psrc[sizeW * 4 + 5]) >> 2;
419  pdest[2] = (((int)psrc[2]) + psrc[6] + psrc[sizeW * 4 + 2] + psrc[sizeW * 4 + 6]) >> 2;
420  pdest[3] = (((int)psrc[3]) + psrc[7] + psrc[sizeW * 4 + 3] + psrc[sizeW * 4 + 7]) >> 2;
421  }
422  }
423  sizeW >>= 1; sizeH >>= 1;
424  }
425  }

427  static void ConvertToSRGB(DWORD * linear)
428  {
429  DWORD drgb[3];
430  for (int k = 0; k < 3; k++)
431  {
432  float rgb = ((float)((*linear >> (k * 8)) & 0xff)) / 255.0f;
433  rgb = pow(rgb, 2.2f);
434  drgb[k] = (DWORD)(rgb * 255.0f);
435  }
436  *linear = (*linear & 0xff000000) + (drgb[2] << 16) + (drgb[1] << 8) + (drgb[0] << 0);
437  }

439  void AutoFillTexture(int autoFillData)
440  {
441  DWORD * pix = (DWORD *)malloc(sizeof(DWORD) * SizeW * SizeH);
442  for (int j = 0; j < SizeH; j++)
443  for (int i = 0; i < SizeW; i++)
444  {
445  DWORD * curr = &pix[j*SizeW + i];
446  switch (autoFillData)
447  {
448  case(AUTO_WALL) : *curr = (((j / 4 & 15) == 0) || (((i / 4 & 15) == 0) && ((((i / 4 & 31) == 0) ^ ((j / 4 >> 4) & 1)) == 0))) ?
449  0xff3c3c3c : 0xffb4b4b4; break;
450  case(AUTO_FLOOR) : *curr = (((i >> 7) ^ (j >> 7)) & 1) ? 0xffb4b4b4 : 0xff505050; break;
451  case(AUTO_CEILING) : *curr = (i / 4 == 0 || j / 4 == 0) ? 0xff505050 : 0xffb4b4b4; break;
452  case(AUTO_WHITE) : *curr = 0xffffffff; break;
453  case(AUTO_GRADE_256) : *curr = 0xff000000 + i*0x010101; break;
454  case(AUTO_GRID) : *curr = (i<4) || (i>(SizeW - 5)) || (j<4) || (j>(SizeH - 5)) ? 0xffffffff : 0xff000000; break;
455  default: *curr = 0xffffffff; break;
456  }
458  }
459  FillTexture(pix);
460  }

464 struct Material
465 {
466  ID3D11VertexShader * D3DVert;
467  ID3D11PixelShader * D3DPix;
468  Texture * Tex;
469  ID3D11InputLayout * InputLayout;
470  UINT VertexSize;
471  ID3D11SamplerState * SamplerState;
472  ID3D11RasterizerState * Rasterizer;
473  ID3D11DepthStencilState * DepthState;
474  ID3D11BlendState * BlendState;
475 
476  enum { MAT_WRAP = 1, MAT_WIRE = 2, MAT_ZALWAYS = 4, MAT_NOCULL = 8 , MAT_TRANS = 16};
477  Material(Texture * t, DWORD flags = MAT_WRAP | MAT_TRANS, D3D11_INPUT_ELEMENT_DESC * vertexDesc = NULL, int numVertexDesc = 3,
478  char* vertexShader = NULL, char* pixelShader = NULL, int vSize = 24) : Tex(t), VertexSize(vSize)
479  {
480  D3D11_INPUT_ELEMENT_DESC defaultVertexDesc[] = {
481  { "Position", 0, DXGI_FORMAT_R32G32B32_FLOAT, 0, 0, D3D11_INPUT_PER_VERTEX_DATA, 0 },
482  { "Color", 0, DXGI_FORMAT_B8G8R8A8_UNORM, 0, 12, D3D11_INPUT_PER_VERTEX_DATA, 0 },
483  { "TexCoord", 0, DXGI_FORMAT_R32G32_FLOAT, 0, 16, D3D11_INPUT_PER_VERTEX_DATA, 0 }, };
484 
485  // Use defaults if no shaders specified
486  char* defaultVertexShaderSrc =
487  "float4x4 ProjView; float4 MasterCol;"
488  "void main(in float4 Position : POSITION, in float4 Color : COLOR0, in float2 TexCoord : TEXCOORD0,"
489  " out float4 oPosition : SV_Position, out float4 oColor: COLOR0, out float2 oTexCoord : TEXCOORD0)"
490  "{ oPosition = mul(ProjView, Position); oTexCoord = TexCoord; "
491  " oColor = MasterCol * Color; }";
492  char* defaultPixelShaderSrc =
493  "Texture2D Texture : register(t0); SamplerState Linear : register(s0); "
494  "float4 main(in float4 Position : SV_Position, in float4 Color: COLOR0, in float2 TexCoord : TEXCOORD0) : SV_Target"
495  "{ float4 TexCol = Texture.Sample(Linear, TexCoord); "
496  " if (TexCol.a==0) clip(-1); " // If alpha = 0, don't draw
497  " return(Color * TexCol); }";
498 
499  if (!vertexDesc) vertexDesc = defaultVertexDesc;
500  if (!vertexShader) vertexShader = defaultVertexShaderSrc;
501  if (!pixelShader) pixelShader = defaultPixelShaderSrc;
502 
503  // Create vertex shader
504  ID3DBlob * blobData;
505  D3DCompile(vertexShader, strlen(vertexShader), 0, 0, 0, "main", "vs_4_0", 0, 0, &blobData, 0);
506  DIRECTX.Device->CreateVertexShader(blobData->GetBufferPointer(), blobData->GetBufferSize(), NULL, &D3DVert);
507 
508  // Create input layout
509  DIRECTX.Device->CreateInputLayout(vertexDesc, numVertexDesc,
510  blobData->GetBufferPointer(), blobData->GetBufferSize(), &InputLayout);
511  blobData->Release();
512 
513  // Create pixel shader
514  D3DCompile(pixelShader, strlen(pixelShader), 0, 0, 0, "main", "ps_4_0", 0, 0, &blobData, 0);
515  DIRECTX.Device->CreatePixelShader(blobData->GetBufferPointer(), blobData->GetBufferSize(), NULL, &D3DPix);
516  blobData->Release();
517 
518  // Create sampler state
519  D3D11_SAMPLER_DESC ss; memset(&ss, 0, sizeof(ss));
520  ss.AddressU = ss.AddressV = ss.AddressW = flags & MAT_WRAP ? D3D11_TEXTURE_ADDRESS_WRAP : D3D11_TEXTURE_ADDRESS_BORDER;
521  ss.Filter = D3D11_FILTER_ANISOTROPIC;
522  ss.MaxAnisotropy = 8;
523  ss.MaxLOD = 15;
524  DIRECTX.Device->CreateSamplerState(&ss, &SamplerState);
525 
526  // Create rasterizer
527  D3D11_RASTERIZER_DESC rs; memset(&rs, 0, sizeof(rs));
528  rs.AntialiasedLineEnable = rs.DepthClipEnable = true;
529  rs.CullMode = flags & MAT_NOCULL ? D3D11_CULL_NONE : D3D11_CULL_BACK;
530  rs.FillMode = flags & MAT_WIRE ? D3D11_FILL_WIREFRAME : D3D11_FILL_SOLID;
531  DIRECTX.Device->CreateRasterizerState(&rs, &Rasterizer);
532 
533  // Create depth state
534  D3D11_DEPTH_STENCIL_DESC dss;
535  memset(&dss, 0, sizeof(dss));
536  dss.DepthEnable = true;
537  dss.DepthFunc = flags & MAT_ZALWAYS ? D3D11_COMPARISON_ALWAYS : D3D11_COMPARISON_LESS;
538  dss.DepthWriteMask = D3D11_DEPTH_WRITE_MASK_ALL;
539  DIRECTX.Device->CreateDepthStencilState(&dss, &DepthState);
540 
541  //Create blend state - trans or otherwise
542  D3D11_BLEND_DESC bm;
543  memset(&bm, 0, sizeof(bm));
544  bm.RenderTarget[0].BlendEnable = flags & MAT_TRANS ? true : false;
545  bm.RenderTarget[0].BlendOp = bm.RenderTarget[0].BlendOpAlpha = D3D11_BLEND_OP_ADD;
546  bm.RenderTarget[0].SrcBlend = bm.RenderTarget[0].SrcBlendAlpha = D3D11_BLEND_SRC_ALPHA;
547  bm.RenderTarget[0].DestBlend = bm.RenderTarget[0].DestBlendAlpha = D3D11_BLEND_INV_SRC_ALPHA;
548  bm.RenderTarget[0].RenderTargetWriteMask = D3D11_COLOR_WRITE_ENABLE_ALL;
549  DIRECTX.Device->CreateBlendState(&bm, &BlendState);
550  }
551  ~Material()
552  {
553  Release(D3DVert);
554  Release(D3DPix);
555  delete Tex; Tex = nullptr;
556  Release(InputLayout);
557  Release(SamplerState);
558  Release(Rasterizer);
559  Release(DepthState);
560  Release(BlendState);
561  }

601  DWORD ModifyColor(DWORD c, XMFLOAT3 pos)
602  {
603  #define GetLengthLocal(v) (sqrt(v.x*v.x + v.y*v.y + v.z*v.z))
604  float dist1 = GetLengthLocal(XMFLOAT3(pos.x - (-2), pos.y - (4), pos.z - (-2)));
605  float dist2 = GetLengthLocal(XMFLOAT3(pos.x - (3), pos.y - (4), pos.z - (-3)));
606  float dist3 = GetLengthLocal(XMFLOAT3(pos.x - (-4), pos.y - (3), pos.z - (25)));
607  int bri = rand() % 160;
608  float R = ((c >> 16) & 0xff) * (bri + 192.0f*(0.65f + 8 / dist1 + 1 / dist2 + 4 / dist3)) / 255.0f;
609  float G = ((c >> 8) & 0xff) * (bri + 192.0f*(0.65f + 8 / dist1 + 1 / dist2 + 4 / dist3)) / 255.0f;
610  float B = ((c >> 0) & 0xff) * (bri + 192.0f*(0.65f + 8 / dist1 + 1 / dist2 + 4 / dist3)) / 255.0f;
611  return( (c & 0xff000000) + ((R>255 ? 255 : (DWORD)R) << 16) + ((G>255 ? 255 : (DWORD)G) << 8) + (B>255 ? 255 : (DWORD)B));
612  }
